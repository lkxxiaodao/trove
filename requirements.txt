# InfoVault 项目依赖
# 建议使用 Python 3.11 ~ 3.12（PySide6 对 3.13 的 DLL 兼容性不稳定）
#
# 创建环境：
#   conda create -n infovault python=3.12
#   conda activate infovault
#   pip install -r requirements.txt

# ===== GUI 框架 =====
PySide6>=6.5,<7.0         # Qt for Python 界面库
qt-material>=2.14          # Material Design 主题（离线使用需打包主题文件）

# ===== Windows 系统交互 =====
pywin32>=306               # Windows API（剪贴板 WM_CLIPBOARDUPDATE 钩子）
keyboard>=0.13.5           # 全局热键注册（备选：pynput）

# ===== 打包（开发依赖） =====
pyinstaller>=6.0           # 生成便携版 exe
